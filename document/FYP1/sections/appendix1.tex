\section{API Endpoint Specifications}

\subsection{POST /api/generate}
\textbf{Description:} Initiates story generation with the specified prompt and configuration parameters.

\subsection{POST /api/generate/voice}
\textbf{Description:} Initiates story generation from voice input using speech-to-text processing.

\textbf{Request Body:}
\begin{verbatim}
{
  "audio_file": "base64_encoded_audio_data",
  "language": "en-US (optional, default: auto-detect)",
  "generate_images": boolean (optional, default: true)
}
\end{verbatim}

\subsection{POST /api/chat}
\textbf{Description:} Interactive chat mode for conversational story development.

\textbf{Request Body:}
\begin{verbatim}
{
  "message": "string (user chat message)",
  "session_id": "string (optional, for continuing conversation)",
  "context": "story_context (optional)"
}
\end{verbatim}

\subsection{POST /api/generate/audio}
\textbf{Description:} Generates audio narration for existing story.

\textbf{Request Body:}
\begin{verbatim}
{
  "story_id": "string (required)",
  "voice_style": "string (optional, default: 'neutral')",
  "language": "string (optional, default: 'en')"
}
\end{verbatim}

\subsection{POST /api/generate/character-consistent}
\textbf{Description:} Generates images with character consistency across scenes.

\textbf{Request Body:}
\begin{verbatim}
{
  "story_id": "string (required)",
  "character_reference": "base64_image_data (optional)",
  "style_transfer": boolean (optional, default: false)
}
\end{verbatim}

\subsection{GET /api/stories}
\textbf{Description:} Retrieves list of user's generated stories with metadata.

\textbf{Query Parameters:}
\begin{verbatim}
?limit=10&offset=0&filter=recent
\end{verbatim}

\subsection{GET /api/stories/{story_id}
\textbf{Description:} Retrieves specific story with all associated media.

\subsection{POST /api/export/pdf}
\textbf{Description:} Exports story to PDF format with custom styling.

\textbf{Request Body:}
\begin{verbatim}
{
  "story_id": "string (required)",
  "format": "string (optional, default: 'standard')",
  "include_images": boolean (optional, default: true)
}
\end{verbatim}

\subsection{POST /api/export/audiobook}
\textbf{Description:} Exports story as audiobook with narration.

\textbf{Request Body:}
\begin{verbatim}
{
  "story_id": "string (required)",
  "voice_style": "string (optional, default: 'neutral')",
  "background_music": boolean (optional, default: false)
}
\end{verbatim}


\subsection{POST /api/evaluate}
\textbf{Description:} Evaluates story quality and provides feedback.

\textbf{Request Body:}
\begin{verbatim}
{
  "story_id": "string (required)",
  "evaluation_criteria": ["creativity", "coherence", "age_appropriateness"]
}
\end{verbatim}

\subsection{GET /api/user/preferences}
\textbf{Description:} Retrieves user preferences and learning data.

\subsection{PUT /api/user/preferences}
\textbf{Description:} Updates user preferences and settings.

\textbf{Request Body:}
\begin{verbatim}
{
  "preferred_style": "string (optional)",
  "age_group": "string (optional)",
  "language": "string (optional)",
  "image_style": "string (optional)"
}
\end{verbatim}

\textbf{Request Headers:}
\begin{verbatim}
Content-Type: application/json
\end{verbatim}

\textbf{Request Body:}
\begin{verbatim}
{
  "prompt": "string (required, 5-500 characters recommended)",
  "generate_images": boolean (optional, default: true)
}
\end{verbatim}

\textbf{Example Request:}
\begin{verbatim}
{
  "prompt": "A cat who learns to sail the seas",
  "generate_images": true
}
\end{verbatim}

\textbf{Success Response (200 OK):}
\begin{verbatim}
{
  "status": "Images generated successfully. ✅",
  "story": {
    "title": "The Sailing Cat",
    "setting": "A peaceful harbor by the ocean",
    "characters": ["Mittens the Cat", "Captain Jim"],
    "scenes": [
      {
        "scene_number": 1,
        "text": "Mittens the cat had always been curious...",
        "image_description": "Orange cat looking at boats in harbor",
        "image_path": "generated/images/sailing_cat_1697..._scene_1.png",
        "image_url": "/generated/images/sailing_cat_1697..._scene_1.png"
      },
      // ... additional scenes
    ],
    "meta": {
      "prompt_type": "normal",
      "enhanced_prompt": "A brave cat learns sailing from captain"
    }
  }
}
\end{verbatim}

\textbf{Error Response (400 Bad Request - Invalid Prompt):}
\begin{verbatim}
{
  "error": "Your prompt appears invalid or meaningless. 
            Please enter a more meaningful idea."
}
\end{verbatim}

\textbf{Error Response (500 Internal Server Error):}
\begin{verbatim}
{
  "detail": "Story generation failed: [specific error message]"
}
\end{verbatim}

\subsection{GET /device}
\textbf{Description:} Returns hardware information including CPU/GPU availability and capabilities.

\textbf{Success Response (200 OK - GPU Available):}
\begin{verbatim}
{
  "device": "cuda",
  "name": "NVIDIA GeForce GTX 1060"
}
\end{verbatim}

\textbf{Success Response (200 OK - CPU Only):}
\begin{verbatim}
{
  "device": "cpu"
}
\end{verbatim}

\subsection{GET /generated/\{path\}}
\textbf{Description:} Static file serving for generated images, stories, and other content.

\textbf{Example URLs:}
\begin{itemize}
    \item \texttt{/generated/images/story\_123\_scene\_1.png} - Scene illustration
    \item \texttt{/generated/stories/story\_title\_123.json} - Story JSON
    \item \texttt{/generated/exports/story\_title\_123.pdf} - PDF export
    \item \texttt{/generated/evaluations/story\_title\_123\_eval.json} - Quality metrics
\end{itemize}

\textbf{Response:} Binary file content with appropriate MIME type

\section{Complete System Requirements}

\subsection{Minimum Hardware Requirements}
\begin{itemize}
    \item \textbf{Processor:} Intel Core i5 (8th gen) or AMD Ryzen 5 equivalent
    \item \textbf{RAM:} 12 GB DDR4 (increased for TTS/STT models)
    \item \textbf{Storage:} 50 GB available space (30 GB for models, 20 GB for generated content)
    \item \textbf{Operating System:} Windows 10/11, macOS 10.15+, Ubuntu 20.04+
    \item \textbf{Network:} Internet connection for initial model download and Ollama installation
    \item \textbf{Audio:} Microphone and speakers for voice features
    \item \textbf{Note:} CPU-only systems functional but slow (3-5 min per image, 2-3 min per audio)
\end{itemize}

\subsection{Recommended Hardware Requirements}
\begin{itemize}
    \item \textbf{Processor:} Intel Core i7 (10th gen) or AMD Ryzen 7 equivalent
    \item \textbf{RAM:} 24 GB DDR4 (for optimal multi-agent performance)
    \item \textbf{GPU:} NVIDIA RTX 3060 12GB or higher (RTX 4080 optimal)
    \item \textbf{VRAM:} 12+ GB dedicated GPU memory (for character consistency and upscaling)
    \item \textbf{Storage:} 100 GB NVMe SSD available space
    \item \textbf{Audio:} High-quality microphone and speakers for voice features
    \item \textbf{Note:} GPU dramatically improves all generation speeds (30-60s vs 3-5min)
\end{itemize}

\section{Complete System Error Codes and Troubleshooting}

\subsection{HTTP Error Codes}
\begin{itemize}
    \item \textbf{400 Bad Request:} Invalid input, malformed request, or validation errors
    \item \textbf{401 Unauthorized:} Authentication required or invalid credentials
    \item \textbf{403 Forbidden:} Insufficient permissions or rate limit exceeded
    \item \textbf{404 Not Found:} Resource not found (story, user, file)
    \item \textbf{409 Conflict:} Resource already exists or concurrent modification
    \item \textbf{413 Payload Too Large:} File size exceeds limits
    \item \textbf{422 Unprocessable Entity:} Invalid data format or business logic errors
    \item \textbf{429 Too Many Requests:} Rate limit exceeded
    \item \textbf{500 Internal Server Error:} Agent processing failure, model inference error, or system error
    \item \textbf{502 Bad Gateway:} External service unavailable
    \item \textbf{503 Service Unavailable:} System maintenance or overload
    \item \textbf{504 Gateway Timeout:} Request timeout
\end{itemize}

\subsection{Agent-Specific Error Codes}

\subsection{Comprehensive Troubleshooting Guide}

\textbf{Model and Agent Issues:}
\begin{itemize}
    \item \textbf{ModuleNotFoundError or ImportError:} Ensure virtual environment is activated and all dependencies installed via \texttt{pip install -r requirements.txt}
    \item \textbf{FileNotFoundError for models:} Verify all model paths in environment variables point to valid files
    \item \textbf{CUDA out of memory:} Close other GPU-intensive applications. System will automatically fall back to CPU
    \item \textbf{Agent processing failed:} Check agent logs, verify model availability, restart services
    \item \textbf{JSON parsing errors:} System includes automatic retry logic. If persistent, try rephrasing input
\end{itemize}

\textbf{Database and Storage Issues:}
\begin{itemize}
    \item \textbf{Database connection failed:} Verify PostgreSQL/Redis is running, check connection strings
    \item \textbf{File storage errors:} Check disk space, verify write permissions, ensure storage backend is configured
    \item \textbf{Session management issues:} Clear Redis cache, restart Redis service
    \item \textbf{User data not persisting:} Check database permissions, verify user authentication
\end{itemize}

\textbf{API and Network Issues:}
\begin{itemize}
    \item \textbf{Connection refused errors:} Verify backend is running on port 8000, check firewall settings
    \item \textbf{CORS errors:} Check CORS configuration, verify allowed origins
    \item \textbf{WebSocket connection failed:} Verify WebSocket server is running, check proxy configuration
    \item \textbf{Rate limiting:} Check rate limit configuration, implement proper queuing
\end{itemize}

\textbf{Frontend Issues:}
\begin{itemize}
    \item \textbf{Blank page:} Check browser console for errors, ensure Next.js dev server is running
    \item \textbf{Images not displaying:} Check network errors, verify file paths, check CORS settings
    \item \textbf{Voice input not working:} Check microphone permissions, verify Web Speech API support
    \item \textbf{Audio playback issues:} Check audio codec support, verify file format compatibility
    \item \textbf{Theme not persisting:} Check localStorage permissions, verify client-side rendering
\end{itemize}

\textbf{Performance Issues:}
\begin{itemize}
    \item \textbf{Slow generation:} Expected on CPU-only systems. Consider GPU upgrade or cloud processing
    \item \textbf{Memory usage high:} Monitor RAM usage, implement model caching, restart services
    \item \textbf{Concurrent user issues:} Implement proper queuing, scale horizontally
    \item \textbf{File generation timeouts:} Increase timeout limits, optimize generation parameters
\end{itemize}

\textbf{Security Issues:}
\begin{itemize}
    \item \textbf{Authentication failures:} Check JWT configuration, verify user credentials
    \item \textbf{File upload security:} Validate file types, implement size limits, scan for malware
    \item \textbf{Input validation:} Sanitize all user inputs, implement proper validation
    \item \textbf{API security:} Implement rate limiting, use HTTPS, validate API keys
\end{itemize}
